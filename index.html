ระบุสถานะรายข้อ

การเชื่อมต่อชุดข้อมูลเข้า Google Sheets (MDRO_AUDIT)

สำหรับจัดเก็บ ตรวจสอบย้อนหลัง และสรุปแผนกต่าง ๆ ในองค์กร

1. ตั้งค่า Webhook URL

หากท่านนำสคริปต์ไปติดตั้งใน Google Sheets ให้นำ URL ที่คัดลอกมาวางในช่องนี้

2. โค้ด Google Apps Script (ฉบับประเมินรวดเร็ว)

1. เข้าสู่ **Google Sheets** ของท่าน เมนู **ส่วนขยาย (Extensions)** > **Apps Script**

2. นำรหัสโค้ดที่ตัดฟิลด์ประชากรออกทั้งหมดตัวนี้ไปวางแทนที่ และกดบันทึก:

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("MDRO_AUDIT");
    
    if (!sheet) {
      sheet = SpreadsheetApp.getActiveSpreadsheet().insertSheet("MDRO_AUDIT");
      sheet.appendRow([
        "วันที่บันทึกระบบ", "วันที่ประเมิน", "อัตราปฏิบัติถูกต้อง %", "รวมข้อปฏิบัติ", "รวมข้อไม่ปฏิบัติ", "รวมข้อไม่เกี่ยวข้อง", "หมายเหตุเพิ่มเติม"
      ]);
    }
    
    sheet.appendRow([
      new Date(),
      params.evalDate || "",
      params.complianceRate || 0,
      params.totalComplied || 0,
      params.totalNotComplied || 0,
      params.totalNA || 0,
      params.generalNotes || ""
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({result: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({result: "error", error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

3. กดปุ่ม **การทำให้ใช้งานได้ (Deploy)** > **เว็บแอปพลิเคชัน (Web app)** ตั้งค่าสิทธิ์เข้าถึงของทุกคน (Anyone) แล้วนำ URL มากรอกช่องด้านบนได้ทันที
